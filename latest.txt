import json, random, subprocess, os

UNIVERSE_SEED = 987654
random.seed(UNIVERSE_SEED)

galaxy = {"systems": []}

for sys_idx in range(5):  # number of systems
    system_seed = random.randint(0, 999999)
    rng = random.Random(system_seed)
    
    system_id = f"system_{sys_idx}"
    
    # Generate stars
    stars = [{"star_id": f"{system_id}_star_0", "type": rng.choice(["G","K","M"]), "seed": rng.randint(0,999999)}]
    
    # Generate planets
    planets = []
    for i in range(rng.randint(1, 8)):
        planet_seed = rng.randint(0, 999999)
        planets.append({
            "planet_id": f"{system_id}_planet_{i}",
            "type": rng.choice(["terrestrial","gas_giant","ice"]),
            "seed": planet_seed,
            "stations": []
        })
    
    # Generate stations per planet
    for planet in planets:
        station_count = rng.randint(0, 3)
        for j in range(station_count):
            station_seed = rng.randint(0, 999999)
            planet["stations"].append({
                "station_id": f"{planet['planet_id']}_station_{j}",
                "type": rng.choice(["trade","military","refinery"]),
                "seed": station_seed
            })
    
    # Generate ships per system
    ships = []
    for k in range(rng.randint(2, 5)):
        ship_seed = rng.randint(0, 999999)
        ship_class = rng.choice(["Corvette","Frigate","Cruiser","Battleship"])
        ships.append({
            "ship_id": f"{system_id}_ship_{k}",
            "class": ship_class,
            "seed": ship_seed
        })
        # Generate ships via Blender headless
        ship_json = {
            "ship_id": f"{system_id}_ship_{k}",
            "class": ship_class,
            "seed": ship_seed,
            "modules": {"engine":"ion_thruster_mk1", "weapon":["laser_mk1"]},
            "hardpoints": rng.randint(2, 6)
        }
        os.makedirs("build/ships", exist_ok=True)
        ship_json_path = f"build/ships/{ship_json['ship_id']}.json"
        with open(ship_json_path, "w") as f:
            json.dump(ship_json, f)
        subprocess.run([
            "blender", "--background", "--python", "generate_ship.py", "--",
            "--ship_json", ship_json_path,
            "--output_dir", "build/ships"
        ])
    
    galaxy["systems"].append({
        "system_id": system_id,
        "seed": system_seed,
        "stars": stars,
        "planets": planets,
        "ships": ships
    })

# Save universe data
with open("build/universe.json", "w") as f:
    json.dump({"universe_seed": UNIVERSE_SEED, "galaxy": galaxy}, f, indent=2)

import bpy, os, json, numpy as np

# -------------------------------
# 1. Generate Planet Mesh from Heightmap
# -------------------------------
def create_planet_mesh(planet_id, heightmap, radius=1.0):
    size = heightmap.shape[0]
    mesh_name = f"{planet_id}_mesh"
    mesh = bpy.data.meshes.new(mesh_name)
    obj = bpy.data.objects.new(mesh_name, mesh)
    bpy.context.collection.objects.link(obj)
    
    verts = []
    faces = []

    # Simple grid-to-sphere projection
    for y in range(size):
        for x in range(size):
            h = heightmap[y, x] * 0.1  # scale height
            theta = (x / size) * 3.14159 * 2
            phi = (y / size) * 3.14159
            vx = (radius + h) * np.sin(phi) * np.cos(theta)
            vy = (radius + h) * np.cos(phi)
            vz = (radius + h) * np.sin(phi) * np.sin(theta)
            verts.append((vx, vy, vz))
    
    for y in range(size-1):
        for x in range(size-1):
            idx = y*size + x
            faces.append((idx, idx+1, idx+size+1, idx+size))
    
    mesh.from_pydata(verts, [], faces)
    mesh.update()
    return obj

# -------------------------------
# 2. Generate Foliage
# -------------------------------
def generate_foliage(obj, foliage_list):
    for f in foliage_list:
        # Create placeholder mesh for tree/rock
        bpy.ops.mesh.primitive_cone_add(vertices=6, radius1=0.01, depth=0.05)
        tree = bpy.context.object
        tree.location = f['position']
        tree.parent = obj

# -------------------------------
def animate_planet(obj, rotation_speed=0.01, orbit_speed=0.001, orbit_radius=1.0):
    # Rotation animation
    obj.rotation_euler[2] = 0
    obj.keyframe_insert(data_path="rotation_euler", frame=0)
    obj.rotation_euler[2] += rotation_speed
    obj.keyframe_insert(data_path="rotation_euler", frame=250)
    
    # Orbit animation along Z axis
    obj.location[0] = orbit_radius
    obj.keyframe_insert(data_path="location", frame=0)
    obj.location[0] = -orbit_radius
    obj.keyframe_insert(data_path="location", frame=250)

# -------------------------------
# 4. Generate LODs
# -------------------------------
def generate_lods(obj, output_dir, lod_count=3):
    lod_files = []
    for i in range(1, lod_count+1):
        ratio = 1.0 / (2**i)
        dec_obj = obj.copy()
        dec_obj.data = obj.data.copy()
        bpy.context.collection.objects.link(dec_obj)
        mod = dec_obj.modifiers.new(name="Decimate", type='DECIMATE')
        mod.ratio = ratio
        bpy.context.view_layer.objects.active = dec_obj
        bpy.ops.object.modifier_apply(modifier=mod.name)
        lod_file = os.path.join(output_dir, f"{obj.name}_LOD{i}.obj")
        bpy.ops.export_scene.obj(filepath=lod_file, use_selection=True)
        lod_files.append(lod_file)
        bpy.data.objects.remove(dec_obj)
    return lod_files

def export_planet(planet_data, output_dir):

def generate_planet(index):
    bpy.ops.mesh.primitive_uv_sphere_add(radius=1)
    planet = bpy.context.object
    planet.name = f"NF_Planet_{index}"

    # Add NovaForge metadata
    planet["nf_seed"] = random.randint(0, 999999)
    planet["nf_density"] = random.uniform(2.0, 6.0)
    planet["nf_meltPoint"] = random.uniform(800, 1500)

    return planet

def export_planet(planet):
    mesh_file = os.path.join(MESH_PATH, planet.name + ".fbx")
    data_file = os.path.join(DATA_PATH, planet.name + ".json")

    bpy.ops.object.select_all(action='DESELECT')
    planet.select_set(True)
    bpy.context.view_layer.objects.active = planet

    bpy.ops.export_scene.fbx(filepath=mesh_file, use_selection=True)

    metadata = {
        "name": planet.name,
        "seed": planet["nf_seed"],
        "density": planet["nf_density"],
        "meltPoint": planet["nf_meltPoint"]
    }

    with open(data_file, "w") as f:
        json.dump(metadata, f, indent=4)

def batch_create(count):
    for i in range(count):
        p = generate_planet(i)
        export_planet(p)

batch_create(10)

class NOVAFORGE_OT_ValidateAssets(bpy.types.Operator):
    bl_idname = "novaforge.validate_assets"
    bl_label = "Validate NovaForge Assets"

    def execute(self, context):
        errors = []

        for obj in bpy.data.objects:
            if obj.type != 'MESH':
                continue

            if not obj.name.startswith("NF_"):
                errors.append(f"{obj.name}: Missing NF_ prefix")

            if obj.scale != (1.0, 1.0, 1.0):
                errors.append(f"{obj.name}: Scale not applied")

            if len(obj.data.polygons) > 50000:
                errors.append(f"{obj.name}: Polycount too high")

            if not obj.data.uv_layers:
                errors.append(f"{obj.name}: Missing UVs")

        if errors:
            self.report({'ERROR'}, "\n".join(errors))
        else:
            self.report({'INFO'}, "All assets valid")

        return {'FINISHED'}

void DamageResolutionSystem(Entity e)
{
    auto& integrity = Get<CStructuralIntegrity>(e);
    auto& damage    = Get<CDamageState>(e);
    auto& material  = Get<CMaterial>(e);

    float total =
        damage.heatDamage      * material.reactivity +
        damage.kineticDamage   * material.hardness +
        damage.corrosiveDamage * material.corrosionFactor +
        damage.radiationDamage * material.volatility;

    integrity.currentIntegrity -= total;

    if (integrity.currentIntegrity <= integrity.fractureThreshold)
        TriggerFracture(e);

    if (integrity.currentIntegrity <= integrity.collapseThreshold)
        TriggerCollapse(e);
}

import bpy

class NOVAFORGE_OT_GenerateLODs(bpy.types.Operator):
    bl_idname = "novaforge.generate_lods"
    bl_label = "Generate LODs"

    def execute(self, context):
        for obj in bpy.data.objects:
            if obj.type != 'MESH' or not obj.name.startswith("NF_"):
                continue
            generate_lods(obj)
        return {'FINISHED'}

def generate_lods(obj):
    ratios = [1.0, 0.5, 0.2]
    for i, ratio in enumerate(ratios):
        lod = obj.copy()
        lod.data = obj.data.copy()
        lod.name = f"{obj.name}_LOD{i}"
        bpy.context.collection.objects.link(lod)
        if i > 0:
            mod = lod.modifiers.new(name="Decimate", type='DECIMATE')
            mod.ratio = ratio

class NFGEN_OT_FinalizeAssets(bpy.types.Operator):
    bl_idname = "nfgen.finalize_assets"
    bl_label = "Finalize Assets"
    
    def execute(self, context):
        root = context.scene.nf_project_root
        if not os.path.exists(root):
            self.report({'ERROR'}, "Project root folder does not exist")
            return {'CANCELLED'}
        
        metadata_list = []
        
        for obj in bpy.context.collection.objects:
            if "nf_seed" in obj:
                # Generate ECS metadata
                meta = {
                    "name": obj.name,
                    "seed": obj["nf_seed"],
                    "location": list(obj.location),
                    "rotation": list(obj.rotation_euler),
                    "scale": list(obj.scale)
                }
                metadata_list.append(meta)
        
        with open(os.path.join(root, "nf_assets_metadata.json"), "w") as f:
            json.dump(metadata_list, f, indent=4)
        
        self.report({'INFO'}, f"Finalized {len(metadata_list)} assets")
        return {'FINISHED'}

    bpy.utils.register_class(NFGEN_OT_FinalizeAssets)

    bpy.utils.unregister_class(NFGEN_OT_FinalizeAssets)

# utils.py
import bpy
import random

def attach_module(parent_obj, module_name, seed, slot=None):
    random.seed(seed)
    mod = bpy.data.objects.new(module_name, bpy.data.meshes.new(f"{module_name}_mesh"))
    bpy.context.collection.objects.link(mod)
    
    # Attach to parent or slot bone
    if slot and parent_obj.type == 'ARMATURE':
        mod.parent = parent_obj
        mod.parent_type = 'BONE'
        mod.parent_bone = slot
    else:
        mod.parent = parent_obj
    return mod

def create_material(obj, seed):
    random.seed(seed)
    mat = bpy.data.materials.new(name=f"{obj.name}_mat")
    mat.use_nodes = True
    bsdf = mat.node_tree.nodes.get("Principled BSDF")
    if bsdf:
        bsdf.inputs["Base Color"].default_value = (random.random(), random.random(), random.random(), 1)
        bsdf.inputs["Metallic"].default_value = random.random()
        bsdf.inputs["Roughness"].default_value = random.random()
    if obj.data.materials:
        obj.data.materials[0] = mat
    else:
        obj.data.materials.append(mat)
    return mat

def generate_hardpoints(obj):
    # Example: create tool hardpoints
    for i in range(2):
        hp = bpy.data.objects.new(f"hp_tool_{i}", None)
        bpy.context.collection.objects.link(hp)
        hp.parent = obj
        hp.location = (0.1*i, 0, 0)
        hp.empty_display_type = 'ARROWS'

def generate_lods(obj, lod_count=3):
    lod_objects = []
    for i in range(lod_count):
        lod = obj.copy()
        lod.data = obj.data.copy()
        lod.name = f"{obj.name}_LOD{i}"
        bpy.context.collection.objects.link(lod)
        # Apply decimation
        modifier = lod.modifiers.new(name="Decimate", type='DECIMATE')
        modifier.ratio = 1.0 / (i+1)
        bpy.context.view_layer.update()
        lod_objects.append(lod)
    return lod_objects

def bake_collision(obj):
    collision = obj.copy()
    collision.data = obj.data.copy()
    collision.name = f"{obj.name}_collision"
    bpy.context.collection.objects.link(collision)
    modifier = collision.modifiers.new(name="Decimate", type='DECIMATE')
    modifier.ratio = 0.1
    collision.display_type = 'WIRE'
    return collision

# utils.py
import bpy, random

def attach_module(parent_obj, module_name, seed, slot=None):
    random.seed(seed)
    mod = bpy.data.objects.new(module_name, bpy.data.meshes.new(f"{module_name}_mesh"))
    bpy.context.collection.objects.link(mod)
    if slot and parent_obj.type == 'ARMATURE':
        mod.parent = parent_obj
        mod.parent_type = 'BONE'
        mod.parent_bone = slot
    else:
        mod.parent = parent_obj
    return mod

def create_material(obj, seed):
    random.seed(seed)
    mat = bpy.data.materials.new(name=f"{obj.name}_mat")
    mat.use_nodes = True
    bsdf = mat.node_tree.nodes.get("Principled BSDF")
    if bsdf:
        bsdf.inputs["Base Color"].default_value = (random.random(), random.random(), random.random(), 1)
        bsdf.inputs["Metallic"].default_value = random.random()
        bsdf.inputs["Roughness"].default_value = random.random()
    if obj.data.materials:
        obj.data.materials[0] = mat
    else:
        obj.data.materials.append(mat)
    return mat

def generate_hardpoints(obj):
    for i in range(2):
        hp = bpy.data.objects.new(f"hp_tool_{i}", None)
        bpy.context.collection.objects.link(hp)
        hp.parent = obj
        hp.location = (0.1*i, 0, 0)
        hp.empty_display_type = 'ARROWS'

def generate_lods(obj, lod_count=3):
    lod_objects = []
    for i in range(lod_count):
        lod = obj.copy()
        lod.data = obj.data.copy()
        lod.name = f"{obj.name}_LOD{i}"
        bpy.context.collection.objects.link(lod)
        modifier = lod.modifiers.new(name="Decimate", type='DECIMATE')
        modifier.ratio = 1.0 / (i+1)
        lod_objects.append(lod)
    obj["nf_lods"] = lod_objects
    return lod_objects

def bake_collision(obj):
    collision = obj.copy()
    collision.data = obj.data.copy()
    collision.name = f"{obj.name}_collision"
    bpy.context.collection.objects.link(collision)
    modifier = collision.modifiers.new(name="Decimate", type='DECIMATE')
    modifier.ratio = 0.1
    collision.display_type = 'WIRE'
    return collision

def assign_animation(obj, action_name, start_frame, end_frame):
    if obj.animation_data is None:
        obj.animation_data_create()
    action = bpy.data.actions.get(action_name)
    if action:
        obj.animation_data.action = action
        obj.animation_data.action.frame_range = (start_frame, end_frame)
    if "nf_animations" not in obj:
        obj["nf_animations"] = {}
    obj["nf_animations"][action_name] = [start_frame, end_frame]

def generate_props_loot_table(seed, location_type):
    random.seed(seed)
    loot_tables = {
        "ship_interior": ["bobblehead", "scrap_crate", "repair_kit"],
        "station_hangar": ["fuel_cell", "toolbox", "turret_module"]
    }
    chosen_props = random.sample(loot_tables.get(location_type, []), k=2)
    for prop_name in chosen_props:
        prop = bpy.data.objects.new(prop_name, bpy.data.meshes.new(f"{prop_name}_mesh"))
        bpy.context.collection.objects.link(prop)
        prop.location = (random.uniform(-1,1), random.uniform(-1,1), random.uniform(0,1))
        prop.rotation_euler = (random.random()*3.14, random.random()*3.14, random.random()*3.14)
        prop["nf_seed"] = seed

def generate_planet(seed):
    random.seed(seed)
    planet = bpy.data.objects.new("Planet_001", bpy.data.meshes.new("planet_mesh"))
    bpy.context.collection.objects.link(planet)
    # Example: simple sphere + noise for terrain
    bpy.ops.object.select_all(action='DESELECT')
    planet.select_set(True)
    bpy.context.view_layer.objects.active = planet
    bpy.ops.mesh.primitive_uv_sphere_add()
    # Assign material
    create_material(planet, seed)
    generate_lods(planet)
    bake_collision(planet)
    planet["nf_seed"] = seed

TEMPLATES_DIR = os.path.join(bpy.utils.user_resource('SCRIPTS'), "addons", "novaforge_generator", "templates")
REMOTE_TEMPLATE_URL = "https://example.com/novaforge/templates/latest_assets.json"

def load_local_templates():
    templates = {}
    for file_name in os.listdir(TEMPLATES_DIR):
        if file_name.endswith(".json"):
            with open(os.path.join(TEMPLATES_DIR, file_name), 'r') as f:
                templates[file_name] = json.load(f)
    return templates

def load_remote_templates():
    try:
        with urllib.request.urlopen(REMOTE_TEMPLATE_URL) as response:
            templates = json.load(response)
        return templates
    except Exception as e:
        print("Failed to fetch remote templates:", e)
        return {}
